<script>
    // 状态管理
    let books = JSON.parse(localStorage.getItem('novel_books') || '[]');
    let currentBookId = null;
    let fontSize = 18;

    // 初始化
    window.onload = () => {
        renderBookList();
        renderLibraryGrid();
        applyTheme('light');
    };

    // --- 核心功能 ---

    function showView(viewName) {
        // 切换视图
        document.querySelectorAll('.content-container').forEach(el => el.classList.remove('active'));
        document.innerHTML = ''; // 清理

        if(viewName === 'library') {
            document.getElementById('library-view').classList.add('active');
            document.getElementById('reader-controls').style.display = 'none';
            document.getElementById('editor-controls').style.display = 'none';
            renderLibraryGrid();
        } else if(viewName === 'reader') {
            document.getElementById('reader-view-container').classList.add('active');
            document.getElementById('reader-controls').style.display = 'flex';
            document.getElementById('editor-controls').style.display = 'none';
            loadBookToReader(currentBookId);
        } else if(viewName === 'editor') {
            document.getElementById('editor-view-container').classList.add('active');
            document.getElementById('reader-controls').style.display = 'none';
            document.getElementById('editor-controls').style.display = 'flex';
            loadBookToEditor(currentBookId);
        }

        // 更新侧边栏按钮状态
        document.querySelectorAll('.nav-btn').forEach(btn => {
            btn.classList.remove('active');
            if(btn.innerText.includes(viewName === 'library' ? '书架' : '')) {
                // 逻辑简化，实际应用中应更精确
            }
        });
    }

    function createNewBook() {
        const newBook = {
            id: Date.now(),
            title: '未命名作品',
            content: '',
            updatedAt: new Date().toLocaleString()
        };
        books.unshift(newBook);
        saveToLocalStorage();
        currentBookId = newBook.id;
        showView('editor');
        showToast('已创建新作品');
    }

    function loadBookToReader(id) {
        const book = books.find(b => b.id === id);
        if (book) {
            currentBookId = id;
            document.getElementById('reader-view').innerHTML = `<h1>${book.title}</h1><br>${book.content.replace(/
/g, '<br>')}`;
showView('reader');
}
}

    function loadBookToEditor(id) {
        const book = books.find(b => b.id === id);
        if (book) {
            currentBookId = id;
            document.getElementById('editor-title').value = book.title;
            document.getElementById('editor-body').value = book.content;
            showView('editor');
        }
    }

    function saveCurrentBook() {
        const bookIndex = books.findIndex(b => b.id === currentBookId);
        if (bookIndex > -1) {
            books[bookIndex].title = document.getElementById('editor-title').value || '未命名作品';
            books[bookIndex].content = document.getElementById('editor-body').value;
            books[bookIndex].updatedAt = new Date().toLocaleString();
            saveToLocalStorage();
            showToast('保存成功！');
            renderBookList();
            renderLibraryGrid();
        }
    }

    function deleteBook(id, event) {
        event.stopPropagation();
        if(confirm('确定要删除这部作品吗？')) {
            books = books.filter(b => b.id !== id);
            saveToLocalStorage();
            renderBookList();
            renderLibraryGrid();
            showToast('已删除');
        }
    }

    // --- UI 渲染 ---

    function renderBookList() {
        const listContainer = document.getElementById('book-list');
        listContainer.innerHTML = '';
        books.forEach(book => {
            const div = document.createElement('div');
            div.className = 'book-item';
            div.innerHTML = `
                <span onclick="loadBookToReader(${book.id})">${book.title}</span>
                <span class="delete-btn" onclick="deleteBook(${book.id}, event)">✕</span>
            `;
            listContainer.appendChild(div);
        });
    }

    function renderLibraryGrid() {
        const grid = document.getElementById('library-grid');
        grid.innerHTML = '';
        books.forEach(book => {
            const card = document.createElement('div');
            card.className = 'book-item';
            card.style.cssText = 'background: var(--main-bg); padding: 20px; border: 1px solid var(--border-color); cursor:pointer; flex-direction:column; align-items:center; text-align:center;';
            card.onclick = () => loadBookToReader(book.id);
            card.innerHTML = `
                <div style="width:60px; height:80px; background:var(--accent-color); margin-bottom:10px; border-radius:4px; display:flex; align-items:center; justify-content:center; font-size:30px; color:white;">📖</div>
                <div style="font-weight:bold;">${book.title}</div>
                <div style="font-size:10px; opacity:0.6;">${book.updatedAt}</div>
            `;
            grid.appendChild(card);
        });
    }

    // --- 工具函数 ---

    function saveToLocalStorage() {
        localStorage.setItem('novel_books', JSON.stringify(books));
    }

    function changeTheme(theme) {
        document.body.className = 'theme-' + theme;
        applyTheme(theme);
    }

    function applyTheme(theme) {
        document.getElementById('theme-select').value = theme;
    }

    function changeFontSize(delta) {
        fontSize += delta;
        if(fontSize < 12) fontSize = 12;
        if(fontSize > 40) fontSize = 40;
        document.documentElement.style.setProperty('--font-size', fontSize + 'px');
    }

    function showToast(msg) {
        const toast = document.getElementById('toast');
        toast.innerText = msg;
        toast.style.display = 'block';
        setTimeout(() => { toast.style.display = 'none'; }, 2000);
    }

    function toggleEditor() {
        if(currentBookId) loadBookToEditor(currentBookId);
    }

</script>
