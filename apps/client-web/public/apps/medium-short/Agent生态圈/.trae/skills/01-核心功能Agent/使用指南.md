## 代码中调用

### 基础用法

```javascript
const { run, smart, report, list } = require('./trae-agent-supervisor');

async function main() {
  // 1. 列出所有Agent
  list();
  
  // 2. 直接执行Agent
  await run('agent-generator', 'generate', { name: 'my-agent' });
  
  // 3. 智能执行
  await smart('生成一个测试Agent');
  
  // 4. 查看状态
  report();
}

main().catch(console.error);
```

### 高级用法

```javascript
const { orchestrator, workflow, status } = require('./trae-agent-supervisor');

// 监听事件
orchestrator.on('taskStarted', (task) => {
  console.log(`任务开始: ${task.agent}`);
});

orchestrator.on('taskCompleted', (task) => {
  console.log(`任务完成: ${task.agent} (${task.duration}ms)`);
});

orchestrator.on('taskFailed', (task) => {
  console.log(`任务失败: ${task.agent} - ${task.error}`);
});

// 执行工作流
async function runWorkflow() {
  const result = await workflow('create-agent', { name: 'workflow-agent' });
  console.log(`工作流完成: ${result.steps} 个步骤`);
}

// 获取详细状态
function getStatus() {
  const report = status();
  console.log(`成功率: ${report.summary.successRate}`);
  console.log(`最近任务:`, report.recentTasks);
}
```

### 批量执行

```javascript
const { batch } = require('./trae-agent-supervisor');

async function batchTasks() {
  const tasks = [
    { agent: 'agent-generator', command: 'generate', args: { name: 'agent-1' } },
    { agent: 'plan-agent', command: 'create', args: { name: 'plan-1' } },
    { agent: 'spec-agent', command: 'create', args: { name: 'spec-1' } }
  ];
  
  const results = await batch(tasks);
  console.log(`批量执行完成: ${results.filter(r => r.success).length}/${tasks.length}`);
}
```

## Trae对话集成

### 在Trae中使用

你可以直接在Trae对话中让AI执行Agent命令：

**示例对话：**

> 用户：帮我生成一个名为 "data-processor" 的Agent

> AI：我来帮你生成这个Agent。
> ```javascript
> const { run } = require('./trae-agent-supervisor');
> await run('agent-generator', 'generate', { name: 'data-processor' });
> ```
> ✅ Agent "data-processor" 生成成功！

> 用户：创建一个电商平台的项目计划

> AI：我来创建项目计划。
> ```javascript
> const { smart } = require('./trae-agent-supervisor');
> await smart('创建电商平台项目计划');
> ```
> ✅ 项目计划创建成功！

### 自动执行模式

创建一个自动执行脚本：

```javascript
// auto-execute.js
const { smart, report } = require('./trae-agent-supervisor');

async function executeUserRequest(description) {
  console.log(`📝 用户请求: ${description}`);
  console.log('🤖 正在自动执行...\n');
  
  try {
    const result = await smart(description);
    console.log('\n✅ 执行成功！');
    report();
    return result;
  } catch (error) {
    console.log('\n❌ 执行失败:', error.message);
    throw error;
  }
}

// 使用
const userRequest = process.argv[2];
if (userRequest) {
  executeUserRequest(userRequest);
} else {
  console.log('用法: node auto-execute.js "你的请求"');
}
```

使用：
```bash
node auto-execute.js "生成一个测试Agent"
node auto-execute.js "创建项目计划"
```

## Agent详细说明

### 1. Agent Generator (agent-generator)

**功能**：生成新的Agent，创建5文档范式

**命令**：
- `generate <name>` - 生成单个Agent
- `batch <config>` - 批量生成
- `validate <dir>` - 验证Agent完整性
- `list-templates` - 列出模板

**示例**：
```bash
node trae-agent-supervisor.js run agent-generator generate my-agent
```

**输出**：
- SKILL.md - Agent技能描述
- requirement.md - 需求规格
- design.md - 架构设计
- tasks.md - 任务分解
- checklist.md - 质量检查清单

### 2. Plan Agent (plan-agent)

**功能**：项目规划、Sprint管理

**命令**：
- `create <project>` - 创建项目计划
- `sprint <name>` - 创建Sprint计划
- `roadmap <project>` - 创建产品路线图

**示例**：
```bash
node trae-agent-supervisor.js run plan-agent create "电商平台"
```

### 3. Spec Agent (spec-agent)

**功能**：规格定义、文档生成

**命令**：
- `create <title>` - 创建规格文档
- `api <endpoint>` - 定义API接口
- `model <name>` - 定义数据模型

**示例**：
```bash
node trae-agent-supervisor.js run spec-agent create "用户管理系统"
```

### 4. Debug Agent (debug-agent)

**功能**：错误诊断、修复建议

**命令**：
- `analyze <error>` - 分析错误
- `fix <error>` - 提供修复建议
- `check <dir>` - 代码质量检查

**示例**：
```bash
node trae-agent-supervisor.js run debug-agent analyze "Cannot read property of undefined"
```

### 5. Batch Agent Creator (batch-agent-creator)

**功能**：批量创建Agent

**命令**：
- `batch <config>` - 从配置文件批量创建
- `create <n> <prefix>` - 创建n个Agent

**示例**：
```bash
node trae-agent-supervisor.js run batch-agent-creator create 10 test-agent
```

### 6. Meta Agent (meta-agent)

**功能**：系统监控、管理

**命令**：
- `status` - 查看系统状态
- `agents` - 列出所有Agent
- `optimize` - 优化系统

**示例**：
```bash
node trae-agent-supervisor.js run meta-agent status
```

### 7. AutoPilot Agent (autopilot-agent)

**功能**：自动执行、工作流编排

**命令**：
- `start` - 启动自动驾驶模式
- `workflow <name>` - 执行工作流

**示例**：
```bash
node trae-agent-supervisor.js run autopilot-agent workflow new-project
```

## 工作流使用

### 预定义工作流

#### 1. create-agent (创建Agent)
```javascript
const { workflow } = require('./trae-agent-supervisor');

await workflow('create-agent', { name: 'my-agent' });
// 步骤:
// 1. spec-agent 创建规格文档
// 2. agent-generator 生成Agent
```

#### 2. new-project (新建项目)
```javascript
await workflow('new-project', { projectName: '电商平台' });
// 步骤:
// 1. plan-agent 创建项目计划
// 2. spec-agent 创建规格文档
```

#### 3. debug-fix (调试修复)
```javascript
await workflow('debug-fix', { error: '错误信息' });
// 步骤:
// 1. debug-agent 分析错误
// 2. spec-agent 创建修复规格
```

### 自定义工作流

```javascript
const { orchestrator } = require('./trae-agent-supervisor');

// 定义工作流
const myWorkflow = [
  { agent: 'plan-agent', command: 'create', args: { name: 'phase-1' } },
  { agent: 'spec-agent', command: 'create', args: { name: 'spec-1' } },
  { agent: 'agent-generator', command: 'generate', args: { name: 'helper-agent' } }
];

// 执行工作流
async function executeCustomWorkflow() {
  for (const step of myWorkflow) {
    console.log(`执行: ${step.agent} -> ${step.command}`);
    await orchestrator.invoke(step.agent, step.command, step.args);
  }
}
```

## 监控与日志

### 实时监控

```javascript
const { orchestrator } = require('./trae-agent-supervisor');

// 监听所有事件
orchestrator.on('taskStarted', (task) => {
  console.log(`[${new Date().toISOString()}] 开始: ${task.agent}`);
});

orchestrator.on('taskOutput', ({ taskId, type, content }) => {
  if (type === 'stdout') {
    console.log(`[输出] ${content}`);
  }
});

orchestrator.on('taskCompleted', (task) => {
  console.log(`[完成] ${task.agent}: ${task.duration}ms`);
});

orchestrator.on('taskFailed', (task) => {
  console.log(`[失败] ${task.agent}: ${task.error}`);
});
```

### 日志文件

日志保存在 `logs/` 目录：
- `CoreSystem-YYYY-MM-DD.log` - 核心系统日志
- `TaskQueue-YYYY-MM-DD.log` - 任务队列日志
- `AgentExecutor-YYYY-MM-DD.log` - 执行引擎日志

### 状态报告

```javascript
const { status } = require('./trae-agent-supervisor');

const report = status();
console.log(`
会话: ${report.session}
总任务: ${report.summary.total}
成功率: ${report.summary.successRate}
`);

// 最近任务
report.recentTasks.forEach(task => {
  console.log(`${task.status === 'completed' ? '✅' : '❌'} ${task.agent} (${task.duration}ms)`);
});
```

## API参考

### TraeAgentSupervisor API

#### `run(agent, command, args)`
直接执行Agent命令。

**参数**：
- `agent` (string): Agent名称
- `command` (string): 命令
- `args` (object): 参数

**返回**：Promise<object>

**示例**：
```javascript
await run('agent-generator', 'generate', { name: 'my-agent' });
```

#### `smart(description, args)`
智能执行，自动选择Agent。

**参数**：
- `description` (string): 任务描述
- `args` (object): 额外参数

**返回**：Promise<object>

**示例**：
```javascript
await smart('生成一个测试Agent');
```

#### `workflow(name, context)`
执行预定义工作流。

**参数**：
- `name` (string): 工作流名称
- `context` (object): 上下文参数

**返回**：Promise<object>

**示例**：
```javascript
await workflow('create-agent', { name: 'my-agent' });
```

#### `batch(tasks)`
批量执行多个任务。

**参数**：
- `tasks` (array): 任务数组

**返回**：Promise<array>

**示例**：
```javascript
await batch([
  { agent: 'agent-generator', command: 'generate', args: { name: 'agent-1' } },
  { agent: 'plan-agent', command: 'create', args: { name: 'plan-1' } }
]);
```

#### `report()`
显示状态报告。

**示例**：
```javascript
report();
```

#### `list()`
列出所有可用Agent。

**示例**：
```javascript
list();
```

### AgentOrchestrator API

#### `invoke(agentName, command, args, options)`
调用Agent。

#### `executeWorkflow(name, context)`
执行工作流。

#### `selectAgentForTask(description)`
根据描述选择最佳Agent。

#### `getAgents()`
获取所有Agent信息。

#### `getAllTasks()`
获取所有任务状态。

#### `generateReport()`
生成执行报告。

```javascript
// init-project.js
const { workflow, run, report } = require('./trae-agent-supervisor');

initProject(projectName);
```
